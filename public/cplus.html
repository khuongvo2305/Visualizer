#include <iostream>

using namespace std;

 

void printArray(int arr[], int size) {

                int i;

                for (i = 0; i < size; i++) {

                                cout << arr[i] << " ";

                }

                cout << "\n";

}

 

int partition(int arr[], int low, int high) {

                static int len = 7;

                static int count = 0;

                int pivot = arr[high];

                int i = low - 1;

 

                for (int j = low; j <= high - 1; j++) {

                                if (arr[j] < pivot) {

                                                i++;

                                                swap(arr[i], arr[j]);

                                                printArray(arr, len);

                                }

                }

                swap(arr[i + 1], arr[high]);

                printArray(arr, len);

                return (i + 1);

}

 

void quickSort(int arr[], int low, int high) {

                static int len = high + 1;

                static int count = 0;

                cout << "step " << count++ << ": ";

                printArray(arr, len);

                if (low < high) {

                                int pi = partition(arr, low, high);

                                quickSort(arr, low, pi - 1);

                                quickSort(arr, pi + 1, high);

                }             

}

 

int main() {

                int arr[] = { 3,7,8,4,2,1,5 };

                int n = sizeof(arr) / sizeof(arr[0]);

                printArray(arr, n);

                cout << endl;

                quickSort(arr, 0, n - 1);

                cout << "Sorted array: \n";

                printArray(arr, n);

                system("pause");

                return 0;

}
